# ============================================================================
# Install order matters. Do NOT just `pip install -r requirements.txt`:
# torch, torchvision and mmcv must come from their own wheel indexes (below),
# or pip will try to COMPILE mmcv from source and fail.
#
# Tested, lab-standard stack (OpenMMLab 2.x / mmrotate). It installs as prebuilt
# wheels with no compilation. Constraints that actually matter:
#   - Python 3.10  (3.8-3.11 OK; 3.12+ has NO prebuilt mmcv wheel)
#   - torch 2.1.0  (newest torch with prebuilt mmcv wheels)
#   - numpy < 2    (torch 2.1 is built against numpy 1.x)
# CUDA 11.8 build shown; swap cu118 -> cu121 if your driver needs it.
#
#   # 1. PyTorch (CUDA 11.8)
#   pip install torch==2.1.0 torchvision==0.16.0 \
#       --index-url https://download.pytorch.org/whl/cu118
#
#   # 2. mmcv  (prebuilt wheel for torch 2.1.0 + cu118 -- no source build)
#   pip install mmcv==2.1.0 \
#       -f https://download.openmmlab.com/mmcv/dist/cu118/torch2.1.0/index.html
#
#   # 3. everything else
#   pip install -r requirements.txt
# ============================================================================
